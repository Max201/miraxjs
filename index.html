mirax.js

Lightweight JS-Framework to manage DOM & Easy build one-page applications

Quick Start

<script type="text/javascript" src="mirax.min.js"></script>

DOM / CSS / Forms Operations

mx.done(function(){ // DOM Content loaded
    // Set css property
    mx.dom('body').css('background', 'teal');
    // .. or properties
    mx.dom('body').css({background: 'teal', color: 'crimson'});

    // Delete element(s)
    mx.dom('section#some-section').remove();
    // Create element
    mx.dom('body').create('<b></b>');

    // Form data
    mx.dom('#auth-form').form() // {field1: val1, field2: val2, ...}

    // Submit form
    mx.http({url: '/target-sctipt.php', data: '#auth-form', method: 'POST'});

    // Routing. In this case, if you go on /#!some-url
    // mirax should load response from /home-backend.php
    // and display in div#response
    // (!) Response can be updated before displaying via
    // calling preprocessor
    mx.route('some-url').src('/home-backend.php').final('div#response' [, callable response preprocessor])

    // Utilities

    // Slugify
    mx.slug('Hello world!!! mirax - great') // hello-world_mirax_-_great
});
